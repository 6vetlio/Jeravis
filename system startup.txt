#!/bin/bash
# Jarvis Ollama Startup Script for 2x RTX 4090 Server (48GB VRAM)
# Run this after SSH into instance to check GPU and start Ollama

echo "=== Jarvis Ollama Startup (2x RTX 4090) ==="
echo ""

# Check NVIDIA GPU and drivers
echo "Step 1: Checking NVIDIA GPU and drivers..."
if command -v nvidia-smi &> /dev/null; then
    echo "✓ nvidia-smi found"
    if nvidia-smi > /dev/null 2>&1; then
        echo "✓ NVIDIA drivers working"
        nvidia-smi --query-gpu=name,memory.total,driver_version --format=csv,noheader
        GPU_AVAILABLE=true
    else
        echo "✗ NVIDIA driver mismatch detected"
        echo "  Attempting quick fix..."
        sudo modprobe -r nvidia 2>/dev/null
        sudo modprobe nvidia 2>/dev/null
        sleep 2
        if nvidia-smi > /dev/null 2>&1; then
            echo "✓ NVIDIA drivers fixed and working"
            nvidia-smi --query-gpu=name,memory.total,driver_version --format=csv,noheader
            GPU_AVAILABLE=true
        else
            echo "✗ Driver fix failed. Ollama will use CPU (slower)"
            echo "  To fix properly: sudo apt install -y nvidia-driver-550 && sudo reboot"
            GPU_AVAILABLE=false
        fi
    fi
else
    echo "✗ nvidia-smi not found. No NVIDIA GPU detected."
    echo "  Ollama will run on CPU (much slower)"
    GPU_AVAILABLE=false
fi

echo ""

# Check if Ollama is installed
echo "Step 2: Checking Ollama installation..."
if ! command -v ollama &> /dev/null; then
    echo "✗ Ollama not found. Installing..."
    curl -fsSL https://ollama.com/install.sh | sh
    echo "✓ Ollama installed"
else
    echo "✓ Ollama found"
fi

echo ""

# Stop any existing Ollama processes
echo "Step 3: Stopping any existing Ollama processes..."
sudo systemctl stop ollama 2>/dev/null || true
pkill -9 ollama 2>/dev/null || true
sleep 2
echo "✓ Existing processes stopped"

echo ""

# Start Ollama with optimized settings for 2x RTX 4090
echo "Step 4: Starting Ollama with optimized settings..."
export OLLAMA_HOST=0.0.0.0:14342
export OLLAMA_NUM_GPU=2
export OLLAMA_GPU_LAYERS=99
export CUDA_VISIBLE_DEVICES=0,1
export OLLAMA_NUM_THREAD=64
if [ "$GPU_AVAILABLE" = false ]; then
    echo "  Warning: Running without GPU acceleration"
fi
nohup ollama serve > /tmp/ollama.log 2>&1 &
sleep 5

# Check if Ollama is running
echo ""
echo "Step 5: Verifying Ollama is running..."
if curl -s http://localhost:14342/api/tags > /dev/null 2>&1; then
    echo "✓ Ollama is running on http://0.0.0.0:14342"
    
    # Show available models
    echo ""
    echo "Available models:"
    curl -s http://localhost:14342/api/tags | grep -o '"name":"[^"]*"' | cut -d'"' -f4 | while read model; do
        echo "  - $model"
    done
    echo ""
    echo "Note: DeepSeek-R1 models have built-in reasoning/thinking capabilities"
else
    echo "✗ Ollama failed to start"
    echo "Check logs: cat /tmp/ollama.log"
    exit 1
fi

echo ""
echo "=== Startup Complete ==="
echo ""
echo "Status:"
if [ "$GPU_AVAILABLE" = true ]; then
    echo "  GPU:    2x RTX 4090 (48GB VRAM) - Enabled and working"
else
    echo "  GPU:    Not available (CPU mode)"
fi
echo "  Ollama: Running on http://0.0.0.0:14342"
echo "  Server: 85.218.235.6:14342"
echo ""
echo "Commands:"
echo "  View logs:    cat /tmp/ollama.log"
echo "  Stop:         pkill ollama"
echo "  Restart:      Run this script again"
echo "  Test:         curl http://localhost:14342/api/tags"
echo ""
